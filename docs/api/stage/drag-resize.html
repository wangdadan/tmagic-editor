  ABSOLUTE = 'absolute',
  /** 固定定位布局 */
  FIXED = 'fixed',
  /** 流式布局 */
  SORTABLE = 'sortable',
}

# select

  • 参数：

    • {HTMLElement} el 组件Dom节点
    • {MouseEvent} MouseEvent

  • 返回：

    • {void}

  • 用法：

    将选中框渲染并覆盖到选中的组件Dom节点上方

    当选中的节点是不是absolute时，会创建一个新的节点出来作为拖拽目标
